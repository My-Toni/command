Command install mysql di termux by Toni


[ Ikutin caranya dan urutin ]

$ pkg update && pkg upgrade -y

$ termux-setup-storage

$ pkg install php

$ pkg install mariadb

$ pkg install nano

$ mysqld_safe -u root

[ Buat session baru lalu ketik di bawah ini ]

$ mysql -u root

[ Setelah kalian masuk ke mysql nya silahkan ketik di bawah ini ]

$ use mysql

[ Set password mysql nya ]

$ set password for 'root'@'localhost' = password('input password disini bang');

$ flush privileges;

$ exit;

[ setelah itu kalian install PhpMyadmin ]

$ apt search phpmyadmin

$ apt install phpmyadmin

[ Kalau sudah tinggal set folder phpmyadmin ]

$ cd $PREFIX/share/phpmyadmin

$ ls

[ Lalu fix bug login nya, ketik di bawah ini ]

$ nano config.inc.php ( cari cfg[server][$i][host] = 'localhost'; )

$ habis itu ganti 'localhost' jadi '127.0.0.1' lalu simpan

$ untuk menyimpan klik "CTRL + O + ENTER" dan untuk keluar "CTRL + X"

[ Jalankan servernya ]

$ "php -S localhost:1212" atau "php -S 127.0.0.1:1212"

[ lalu buka link nya melalui browser/chrome ]

$ localhost:1212
